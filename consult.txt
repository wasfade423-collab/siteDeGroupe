    <script>
        async function chargerGroupes() {
            try {
                const response = await fetch('TON_URL_API/groupes');
                const groupes = await response.json();
                
                const container = document.getElementById('groupContainer');
                container.innerHTML = ""; 

                groupes.forEach((groupe, index) => {
                    container.innerHTML += formaterCarteGroupe(groupe, index + 1);
                });
            } catch (error) {
                console.error("Erreur de chargement");
                // Optionnel : afficher ton modal d'erreur ici
            }
        }

        function formaterCarteGroupe(groupe, numero) {
            const nbMembres = groupe.membres.length;
            
            const listeMembres = groupe.membres.map((membre, i) => {
                // Utilisation de Amber-100 pour le fond et Amber-700 pour le texte/étoile
                const estPremier = i === 0 ? 
                    `<span class="ml-2 text-[10px] bg-amber-100 text-amber-700 px-2 py-0.5 rounded-full font-bold uppercase tracking-wider">
                        <i class="ri-star-fill text-[10px]"></i> Responsable
                    </span>` : '';
                
                return `
                    <li class="flex items-center py-3 border-b border-slate-100 last:border-0 text-slate-700">
                        <span class="font-mono text-amber-700 font-bold mr-3">${i + 1}.</span>
                        <span class="capitalize">${membre.toLowerCase().trim()}</span>
                        ${estPremier}
                    </li>`;
            }).join('');

            return `
                <div class="bg-white rounded-xl shadow-md hover:shadow-xl transition-all border border-slate-100 overflow-hidden">
                    <div class="bg-slate-100 p-4 border-b border-slate-200 flex justify-between items-center">
                        <span class="font-bold text-slate-700">Groupe n°${numero}</span>
                        <span class="text-xs bg-white text-slate-500 border border-slate-200 px-3 py-1 rounded-full font-bold">
                            ${nbMembres} membres
                        </span>
                    </div>
                    <ul class="p-4 bg-white">
                        ${listeMembres}
                    </ul>
                </div>
            `;
        }

        window.onload = chargerGroupes;
    </script>
